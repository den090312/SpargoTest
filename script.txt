CREATE TABLE Consignment (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    ProductId INTEGER,
    WarehouseId INTEGER
);
CREATE TABLE Pharmacy (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT,
    Address TEXT,
    PhoneNumber TEXT
);
CREATE TABLE Product (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT
);
CREATE TABLE Warehouse (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    PharmacyId INTEGER,
    Name TEXT
);
